class Friends extends React.Component{
  state = { 
    show: true
  }

  CambiarEstado = () =>{
    this.setState({show: !this.state.show})
  }

  render(){
    if(this.state.show){
      return(
        <div id="number1">
        <h1>{this.props.subtitle}</h1>
        <p>{this.props.mytext}</p>
        <button onClick={this.CambiarEstado}> Ocultar </button>
        </div>
      )
    }
    else{
      return <div> No hay elementos 
      <button onClick={this.CambiarEstado}> Mostrar </button>
      </div>
    }
  }
}


function App() {
  return (
    <div>
      <Friends subtitle="Codificando" mytext="Acá va el texto"/>
      <Friends subtitle="Basic" mytext="fisrt text"/>
      </div>
  );
}
